    <article class="markdown-body entry-content" itemprop="mainContentOfPage"><h1><a id="user-content-sba-i2c-ipcore" class="anchor" href="#sba-i2c-ipcore" aria-hidden="true"><span class="octicon octicon-link"></span></a><strong>SBA I2C IPCore</strong></h1>

<hr>

<p><a href="/mriscoc/SBA-Library/blob/Branch_master_0/I2C/image.png" target="_blank"><img src="/mriscoc/SBA-Library/raw/Branch_master_0/I2C/image.png" alt="" style="max-width:100%;"></a></p>

<p>version 1.0 20150526   </p>

<p>I2C is an IPCore designed to send data using the <a href="http://www.nxp.com/documents/user_manual/UM10204.pdf" title="I2C Bus Specification">I2C Bus Specification</a>.  </p>

<p><strong>Author:</strong><br>
(c) Juan S. Vega Martinez<br>
email: <a href="mailto:juan.vega25@gmail.com">juan.vega25@gmail.com</a>   </p>

<p><strong>Notes</strong><br>
v1.0 SBA 1.0 compliant  </p>

<div class="highlight highlight-vhdl"><pre><span class="pl-k">entity</span> <span class="pl-en">I2C_Adapter</span> <span class="pl-k">is</span>
<span class="pl-k">generic</span> (FREQ_SCL: <span class="pl-c1">natural</span><span class="pl-k">:=</span> <span class="pl-c1">100e3</span>); 
<span class="pl-k">port</span> (
  <span class="pl-c">-- SBA Bus Interface</span>
  CLK_I : <span class="pl-k">in</span> <span class="pl-c1">std_logic</span>;
  RST_I : <span class="pl-k">in</span> <span class="pl-c1">std_logic</span>;
  WE_I  : <span class="pl-k">in</span> <span class="pl-c1">std_logic</span>;
  STB_I : <span class="pl-k">in</span> <span class="pl-c1">std_logic</span>;
  ACK_O : <span class="pl-k">out</span> <span class="pl-c1">std_logic</span>;
  ADR_I : <span class="pl-k">in</span> <span class="pl-c1">std_logic_vector</span>(<span class="pl-c1">1</span> <span class="pl-k">downto</span> <span class="pl-c1">0</span>);
  DAT_I : <span class="pl-k">in</span> DATA_Type;
  DAT_O : <span class="pl-k">out</span> DATA_Type;
  <span class="pl-c">-- I2C Interface</span>
  I2C_SCL : <span class="pl-k">inout</span> <span class="pl-c1">std_logic</span>;
  I2C_SDA : <span class="pl-k">inout</span> <span class="pl-c1">std_logic</span>
  );
<span class="pl-k">end</span> <span class="pl-en">I2C_Adapter</span>;</pre></div>

<p><strong>Basic Registers in I2C_Adapter</strong></p>

<div class="highlight highlight-vhdl"><pre>   <span class="pl-k">constant</span> I2C_TXSTA   : <span class="pl-c1">integer</span><span class="pl-k">:=</span><span class="pl-c1">15</span>;                      <span class="pl-c">--Tx Ready</span>
   <span class="pl-k">constant</span> I2C_RXSTA   : <span class="pl-c1">integer</span><span class="pl-k">:=</span><span class="pl-c1">14</span>;                      <span class="pl-c">--Rx Ready</span>
   <span class="pl-k">constant</span> I2C_ReadMEMORY   : <span class="pl-c1">unsigned</span>(<span class="pl-c1">7</span> <span class="pl-k">downto</span> <span class="pl-c1">0</span>)<span class="pl-k">:=</span><span class="pl-c1">x"08"</span>; <span class="pl-c">--Enable Read Data from internal memory</span>
   <span class="pl-k">variable</span> I2C_DEVICE_ADR   : <span class="pl-c1">unsigned</span>(<span class="pl-c1">7</span> <span class="pl-k">downto</span> <span class="pl-c1">0</span>);        <span class="pl-c">--I2C Address Device</span>
   <span class="pl-k">variable</span> I2C_DAT          : <span class="pl-c1">unsigned</span>(<span class="pl-c1">7</span> <span class="pl-k">downto</span> <span class="pl-c1">0</span>);        <span class="pl-c">--Data to Write or Read</span>
   <span class="pl-k">variable</span> I2C_ADR_REG      : <span class="pl-c1">unsigned</span>(<span class="pl-c1">7</span> <span class="pl-k">downto</span> <span class="pl-c1">0</span>);        <span class="pl-c">--Initial address register to Write or Read </span>
   <span class="pl-k">variable</span> I2C_NumberBytesRead: <span class="pl-c1">unsigned</span>(<span class="pl-c1">7</span> <span class="pl-k">downto</span> <span class="pl-c1">0</span>);      <span class="pl-c">--Number of bytes to Write or Read</span>
   <span class="pl-k">variable</span> I2C_ADR_MEMORY   : <span class="pl-c1">unsigned</span>(<span class="pl-c1">7</span> <span class="pl-k">downto</span> <span class="pl-c1">0</span>);        <span class="pl-c">--Address Internal Memory to Multiple Write </span>
   <span class="pl-k">variable</span> I2CFlg           : <span class="pl-c1">std_logic</span>;                   <span class="pl-c">--Flag status </span></pre></div>

<p><strong>Abstract of functions avaliable in the I2C_Adapter</strong>    </p>

<p><strong>1. Single write function.</strong> Write a single data to I2C device.</p>

<p>Example:</p>

<div class="highlight highlight-vhdl"><pre><span class="pl-c">-- Turn On ADXL345</span>
<span class="pl-k">=&gt;</span> I2C_DEVICE_ADR<span class="pl-k">:=</span><span class="pl-c1">x"53"</span>;   <span class="pl-c">-- Set Address Device</span>
   I2C_ADR_REG<span class="pl-k">:=</span><span class="pl-c1">x"2D"</span>;      <span class="pl-c">-- Register 0x2D—POWER_CTL                 </span>
   I2C_DAT<span class="pl-k">:=</span><span class="pl-c1">x"08"</span>;          <span class="pl-c">-- Enable the measurement mode of ADXL345 in the register 0x2D      </span>
   SBAcall(I2CWritebyte);   <span class="pl-c">-- Call I2CWritebyte Routine   </span></pre></div>

<p><strong>2. Single Read function.</strong> Read single data from I2C device.</p>

<p>Example:</p>

<div class="highlight highlight-vhdl"><pre><span class="pl-c">-- Read Device ID ADXL345</span>
<span class="pl-k">=&gt;</span> I2C_DEVICE_ADR <span class="pl-k">:=</span><span class="pl-c1">x"53"</span>;      <span class="pl-c">-- Set Address Device</span>
   I2C_ADR_REG<span class="pl-k">:=</span><span class="pl-c1">x"00"</span>;          <span class="pl-c">-- Register 0x00-Device ID </span>
   SBAcall(I2CReadbyte);        <span class="pl-c">-- Call I2CReadbyte Routine.The read value is returned in the variable I2C_DAT. </span>
<span class="pl-k">=&gt;</span> SBAWrite(GPIO, <span class="pl-c1">x"00"</span> <span class="pl-k">&amp;</span> I2C_DAT);<span class="pl-c">-- Displaying the value of I2C_DAT to the LEDs </span>
</pre></div>

<p><strong>3. Multiple Write function.</strong> Write a Multiple data to an I2C device. It use an internal memory, where the user  can
load all data, using the function I2CLoadbyteToMEMORY and It can upload until 256 data.
The process requires that user define the I2C_DEVICE_ADR(I2C Address Device) and I2C_ADR_REG(Address Register).
The multi-function writing data sequentially writes the addresses of I2C registers starting from the register address I2C_ADR_REG.    </p>

<p>Example:</p>

<div class="highlight highlight-vhdl"><pre><span class="pl-c">-- Multiple Write registers of ADXL345. </span>
<span class="pl-c">-- Writing three data.</span>
<span class="pl-k">=&gt;</span> SBAcall(I2Cwait);                <span class="pl-c">-- Is the I2C module avaliable?</span>
<span class="pl-k">=&gt;</span> I2C_ADR_MEMORY<span class="pl-k">:=</span><span class="pl-c1">x"00"</span>;           <span class="pl-c">-- Reset address internal memory.</span>
   I2C_DAT<span class="pl-k">:=</span><span class="pl-c1">x"10"</span>;                  <span class="pl-c">-- First data</span>
   SBAcall(I2CLoadbyteToMEMORY);    <span class="pl-c">-- Load data within internal memory</span>
<span class="pl-k">=&gt;</span> I2C_DAT<span class="pl-k">:=</span><span class="pl-c1">x"02"</span>;                  <span class="pl-c">-- Second data</span>
   SBAcall(I2CLoadbyteToMEMORY);    <span class="pl-c">-- Load data within internal memory</span>
<span class="pl-k">=&gt;</span> I2C_DAT<span class="pl-k">:=</span><span class="pl-c1">x"25"</span>;                  <span class="pl-c">-- Third data</span>
   SBAcall(I2CLoadbyteToMEMORY);    <span class="pl-c">-- Load data within internal memory</span>
<span class="pl-k">=&gt;</span> I2C_DEVICE_ADR <span class="pl-k">:=</span><span class="pl-c1">x"53"</span>;          <span class="pl-c">-- Set Address Device</span>
   I2C_ADR_REG<span class="pl-k">:=</span><span class="pl-c1">x"1E"</span>;              <span class="pl-c">-- Register Initial</span>
   SBAcall(I2CWritebytes);          <span class="pl-c">-- Call routine I2CWriteBytes </span>
                                <span class="pl-c">-- Address  Value</span>
                                <span class="pl-c">--  0x1E  &lt;- 10</span>
                                <span class="pl-c">--  0x1F  &lt;- 02</span>
                                <span class="pl-c">--  0x20  &lt;- 25</span></pre></div>

<p><strong>4. Multiple Read function.</strong> Read a Multiple data to an I2C device. It use an internal memory, where the user  can
load all data, using the function I2CLoadbyteToMEMORY and It can upload until 256 data.
The process requires that user define the I2C_DEVICE_ADR(I2C Address Device) and I2C_ADR_REG(Address Register).
The function of reading multiple data sequentially reads the addresses of I2C registers starting from the register address I2C_ADR_REG.    </p>

<p>Example:</p>

<div class="highlight highlight-vhdl"><pre><span class="pl-k">=&gt;</span> SBAcall(I2Cwait);             <span class="pl-c">-- Is the I2C module avaliable?</span>
<span class="pl-k">=&gt;</span> I2C_DEVICE_ADR <span class="pl-k">:=</span><span class="pl-c1">x"53"</span>;       <span class="pl-c">-- Set Address Device</span>
   I2C_ADR_REG<span class="pl-k">:=</span><span class="pl-c1">x"32"</span>;           <span class="pl-c">-- Set Register to starting to Read </span>
<span class="pl-c">-- /L:ReadBucle</span>
<span class="pl-k">=&gt;</span> I2C_NumberBytesRead<span class="pl-k">:=</span><span class="pl-c1">x"06"</span>;   <span class="pl-c">-- Numbers bytes to Read</span>
   SBACall(I2CReadbytes);        <span class="pl-c">-- Call I2CReadBytes Routine</span>
<span class="pl-k">=&gt;</span> SBARead(I2C);
<span class="pl-k">=&gt;</span> SBAwait;  reg1<span class="pl-k">:=</span> <span class="pl-c1">x"00"</span> <span class="pl-k">&amp;</span> dati(<span class="pl-c1">7</span> <span class="pl-k">downto</span> <span class="pl-c1">0</span>);  <span class="pl-c">-- first data    X (LSB)</span>

<span class="pl-k">=&gt;</span> SBAwait;  reg3<span class="pl-k">:=</span> <span class="pl-c1">x"00"</span> <span class="pl-k">&amp;</span> dati(<span class="pl-c1">7</span> <span class="pl-k">downto</span> <span class="pl-c1">0</span>);  <span class="pl-c">-- third data    Y (LSB)</span>

<span class="pl-k">=&gt;</span> SBAwait;  reg6<span class="pl-k">:=</span> <span class="pl-c1">x"00"</span> <span class="pl-k">&amp;</span> dati(<span class="pl-c1">7</span> <span class="pl-k">downto</span> <span class="pl-c1">0</span>);  <span class="pl-c">-- sixth data    Z (MSB)</span>
<span class="pl-k">=&gt;</span> SBARead(GPIO2);                             <span class="pl-c">-- Read the status of the switches</span>
<span class="pl-k">=&gt;</span> reg7<span class="pl-k">:=</span> dati(<span class="pl-c1">2</span> <span class="pl-k">downto</span> <span class="pl-c1">0</span>);      
<span class="pl-k">=&gt;</span> <span class="pl-k">case</span> reg7 <span class="pl-k">is</span>                                <span class="pl-c">-- Evaluating the status of the switches</span>
      <span class="pl-k">when</span> <span class="pl-c1">"000"</span> <span class="pl-k">=&gt;</span> SBAWrite(GPIO,reg1);       <span class="pl-c">-- Displaying X(LSB) data to the LEDs</span>
      <span class="pl-k">when</span> <span class="pl-c1">"001"</span> <span class="pl-k">=&gt;</span> SBAWrite(GPIO,reg2);       <span class="pl-c">-- Displaying X(MSB) data to the LEDs</span>
      <span class="pl-k">when</span> <span class="pl-c1">"010"</span> <span class="pl-k">=&gt;</span> SBAWrite(GPIO,reg3);       <span class="pl-c">-- Displaying Y(LSB) data to the LEDs</span>
      <span class="pl-k">when</span> <span class="pl-c1">"011"</span> <span class="pl-k">=&gt;</span> SBAWrite(GPIO,reg4);       <span class="pl-c">-- Displaying Y(MSB) data to the LEDs</span>
      <span class="pl-k">when</span> <span class="pl-c1">"100"</span> <span class="pl-k">=&gt;</span> SBAWrite(GPIO,reg5);       <span class="pl-c">-- Displaying Z(LSB) data to the LEDs</span>
      <span class="pl-k">when</span> <span class="pl-c1">"101"</span> <span class="pl-k">=&gt;</span> SBAWrite(GPIO,reg6);       <span class="pl-c">-- Displaying Z(MSB) data to the LEDs</span>
